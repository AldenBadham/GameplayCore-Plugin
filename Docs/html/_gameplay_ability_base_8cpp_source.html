58bool UGameplayAbilityBase::DoesAbilitySatisfyTagRequirements(const UAbilitySystemComponent& AbilitySystemComponent, const FGameplayTagContainer* SourceTags, const FGameplayTagContainer* TargetTags, FGameplayTagContainer* OptionalRelevantTags) const
59{
60 bool bBlocked = false;
61 bool bMissing = false;
62
63 const UAbilitySystemGlobalsBase& AbilitySystemGlobals = UAbilitySystemGlobalsBase::Get();
64 const FGameplayTag& BlockedTag = AbilitySystemGlobals.ActivateFailTagsBlockedTag;
65 const FGameplayTag& MissingTag = AbilitySystemGlobals.ActivateFailTagsMissingTag;
66
67 // Check if any of this ability's tags are currently blocked
68 if (AbilitySystemComponent.AreAbilityTagsBlocked(GetAssetTags()))
69 {
70 bBlocked = true;
71 }
72
73 // Check to see the required/blocked tags for this ability
74 if (ActivationBlockedTags.Num() || ActivationRequiredTags.Num())
75 {
76 static FGameplayTagContainer AbilitySystemComponentTags;
77 AbilitySystemComponentTags.Reset();
78
79 AbilitySystemComponent.GetOwnedGameplayTags(AbilitySystemComponentTags);
80 AbilitySystemComponent.GetBlockedAbilityTags(AbilitySystemComponentTags);
81
82 if (AbilitySystemComponentTags.HasAny(ActivationBlockedTags))
83 {
84 bBlocked = true;
85 }
86
87 if (!AbilitySystemComponentTags.HasAll(ActivationRequiredTags))
88 {
89 bMissing = true;
90 }
91 }
92
93 // Check source tags
94 if (SourceTags != nullptr)
95 {
96 if (SourceBlockedTags.Num() || SourceRequiredTags.Num())
97 {
98 if (SourceTags->HasAny(SourceBlockedTags))
99 {
100 bBlocked = true;
101 }
102
103 if (!SourceTags->HasAll(SourceRequiredTags))
104 {
105 bMissing = true;
106 }
107 }
108 }
109
110 // Check target tags
111 if (TargetTags != nullptr)
112 {
113 if (TargetBlockedTags.Num() || TargetRequiredTags.Num())
114 {
115 if (TargetTags->HasAny(TargetBlockedTags))
116 {
117 bBlocked = true;
118 }
119
120 if (!TargetTags->HasAll(TargetRequiredTags))
121 {
122 bMissing = true;
123 }
124 }
125 }
126
127 // We succeeded if there were no blocked tags and no missing required tags
128 return !bBlocked && !bMissing;
129}

136void UGameplayAbilityBase::TryActivateAbilityOnSpawn(const FGameplayAbilityActorInfo* ActorInfo, const FGameplayAbilitySpec& Spec) const
137{
138 const bool bIsPredicting = Spec.GetPrimaryInstance()->GetCurrentActivationInfo().ActivationMode == EGameplayAbilityActivationMode::Predicting;
139
140 // Try to activate if activation policy is OnSpawn.
141 if (ActorInfo && !Spec.IsActive() && !bIsPredicting && ActivationPolicy == EAbilityActivationPolicy::OnSpawn)
142 {
143 UAbilitySystemComponent* AbilitySystemComp = ActorInfo->AbilitySystemComponent.Get();
144 const AActor* AvatarActor = ActorInfo->AvatarActor.Get();
145
146 if (IsValid(AbilitySystemComp) && IsValid(AvatarActor))
147 {
148 // Don't try to activate the ability if the avatar is being destroyed and no longer replicated
149 if (!AvatarActor->GetTearOff() && AvatarActor->GetLifeSpan() <= 0.0f)
150 {
151 // Activate the ability only if we're on the authority
152 const bool bIsLocalExecution = NetExecutionPolicy == EGameplayAbilityNetExecutionPolicy::LocalPredicted || NetExecutionPolicy == EGameplayAbilityNetExecutionPolicy::LocalOnly;
153 const bool bIsServerExecution = NetExecutionPolicy == EGameplayAbilityNetExecutionPolicy::ServerOnly || NetExecutionPolicy == EGameplayAbilityNetExecutionPolicy::ServerInitiated;
154
155 const bool bClientShouldActivate = ActorInfo->IsLocallyControlled() && bIsLocalExecution;
156 const bool bServerShouldActivate = ActorInfo->IsNetAuthority() && bIsServerExecution;
157
158 if (bClientShouldActivate || bServerShouldActivate)
159 {
160 AbilitySystemComp->TryActivateAbility(Spec.Handle);
161 }
162 }
163 }
164 }
