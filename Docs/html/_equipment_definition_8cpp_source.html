15void UEquipmentDefinition::PostEditChangeProperty(FPropertyChangedEvent& PropertyChangedEvent)
16{
17 Super::PostEditChangeProperty(PropertyChangedEvent);
18
19 // Check if the Fragments property has changed
20 if (PropertyChangedEvent.Property && PropertyChangedEvent.GetPropertyName() == GET_MEMBER_NAME_CHECKED(UEquipmentDefinition, Fragments))
21 {
22 TSet<UClass*> FragmentClasses;
23 for (int32 i = Fragments.Num() - 1; i >= 0; --i)
24 {
25 if (const UEquipmentFragment* Fragment = Fragments[i]; IsValid(Fragment) && FragmentClasses.Contains(Fragment->GetClass()))
26 {
27 // Remove duplicate fragments
28 Fragments.RemoveAt(i);
29
30 UE_LOG(LogTemp, Warning, TEXT("Only one fragment per class is allowed. Duplicate fragment removed."));
31 }
32 else if (Fragment)
33 {
34 FragmentClasses.Add(Fragment->GetClass());
35 }
36 }
37 }
38}
