113void IGameInitStateInterface::BindOnActorInitStateChanged(FName FeatureName, FGameplayTag RequiredState, bool bCallIfReached)
114{
115 UObject* ThisObject = Cast<UObject>(this);
116 AActor* OwningActor = GetOwningActor();
117
118 UGameInitStateSubsystem* Manager = UGameInitStateSubsystem::GetForActor(OwningActor);
119
120 if (ensure(OwningActor && Manager))
121 {
122 // Bind as a weak lambda because this is not a UObject but is guaranteed to be valid as long as ThisObject is
123 FActorInitStateChangedDelegate Delegate = FActorInitStateChangedDelegate::CreateWeakLambda(ThisObject,
124 [this](const FActorInitStateChangedParams& Params)
125 {
126 this->OnActorInitStateChanged(Params);
127 });
128
129 ActorInitStateChangedHandle = Manager->RegisterAndCallForActorInitState(OwningActor, FeatureName, RequiredState, MoveTemp(Delegate), bCallIfReached);
130 }
131}
