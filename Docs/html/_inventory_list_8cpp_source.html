66FInventoryAddResult FInventoryList::Add(const TSubclassOf<UItemDefinition>& DefinitionClass, const int32 Count)
67{
68 FInventoryAddResult Result;
69
70 if (!IsValid(DefinitionClass) || !OwnerComponent)
71 {
72 UE_LOG(LogInventorySystem, Warning, TEXT("Tried to instantiate an item instance from invalid definition class."));
73 Result.FailureReason = InventorySystemGameplayTags::TAG_Inventory_Failure_NullDefinition;
74 return Result;
75 }
76
77 if (const AActor* OwnerActor = OwnerComponent->GetOwner(); !OwnerActor->HasAuthority())
78 {
79 UE_LOG(LogInventorySystem, Warning, TEXT("Tried to instantiate an item instance from invalid definition class."));
80 Result.FailureReason = InventorySystemGameplayTags::TAG_Inventory_Failure_NotAuthority;
81 return Result;
82 }
83
84 // Check validity of the definition instance, and storable fragments
85 if (!CanAdd(DefinitionClass, Result.FailureReason, Count))
86 {
87 return Result;
88 }
89
90 int32 RemainingCount = Count;
91
92 const UItemDefinition* CachedDefinition = OwnerComponent->GetCachedDefinition(DefinitionClass);
93 const UItemFragment_Storable* StorableFragment = CachedDefinition->FindFragmentByClass<UItemFragment_Storable>();
94
95 // Handles stacking if the object is stackable
96 if (StorableFragment->CanStack())
97 {
98 for (int32 Index = 0; Index < Entries.Num() && RemainingCount > 0; ++Index)
99 {
100 FInventoryEntry& Entry = Entries[Index];
101 if (Entry.Instance->DefinitionClass != DefinitionClass)
102 {
103 continue;
104 }
105
106 const int32 FreeCount = StorableFragment->MaxStackCount - Entry.StackCount;
107 const int32 ToAdd = FMath::Min(RemainingCount, FreeCount);
108
109 if (ToAdd > 0)
110 {
111 Entry.StackCount += ToAdd;
112 RemainingCount -= ToAdd;
113
114 Internal_OnEntryChanged(Index, Entry);
115 Entry.LastStackCount = Entry.StackCount;
116 MarkItemDirty(Entry);
117
118 Result.Instances.Add(Entry.Instance);
119 }
120 }
121 }
122
123 // Calculate the number of new stacks required
124 const int32 MaxStackCount = StorableFragment->CanStack() ? StorableFragment->MaxStackCount : 1;
125 const int32 NeededStacks = FMath::DivideAndRoundUp(RemainingCount, MaxStackCount);
126
127 // Create new stacks for the rest
128 for (int32 i = 0; i < NeededStacks && RemainingCount > 0; ++i)
129 {
130 if (!CanAdd(DefinitionClass, Result.FailureReason, RemainingCount))
131 {
132 break;
133 }
134
135 int32 CurrentCount = RemainingCount;
136 if (UItemInstance* NewInstance = CreateItemInstance(DefinitionClass, CurrentCount))
137 {
138 Result.Instances.Add(NewInstance);
139 RemainingCount = CurrentCount;
140 }
141 }
142
143 return Result;
144}

159FInventoryAddResult FInventoryList::AddItemInstance(UItemInstance* ItemInstance, const int32 Count)
160{
161 FInventoryAddResult Result;
162
163 if (!IsValid(ItemInstance))
164 {
165 Result.FailureReason = InventorySystemGameplayTags::TAG_Inventory_Failure_NullInstance;
166 return Result;
167 }
168
169 // Check if the item can be added
170 if (const TSubclassOf<UItemDefinition> DefinitionClass = ItemInstance->GetDefinitionClass())
171 {
172 if (!CanAdd(DefinitionClass, Result.FailureReason, Count))
173 return Result;
174 }
175
176 // Creating and configuring the new input
177 FInventoryEntry& NewEntry = Entries.AddDefaulted_GetRef();
178 const int32 NewIndex = Entries.Num() - 1;
179
180 NewEntry.Instance = ItemInstance;
181 NewEntry.StackCount = Count;
182 NewEntry.LastStackCount = Count;
183
184 Result.Instances.Add(ItemInstance);
185
186 // Notification du changement
187 Internal_OnEntryAdded(NewIndex, NewEntry);
188 MarkItemDirty(NewEntry);
189
190 return Result;
191}

296UItemInstance* FInventoryList::CreateItemInstance(const TSubclassOf<UItemDefinition>& DefinitionClass, int32& Count)
297{
298 AActor* OwnerActor = OwnerComponent->GetOwner();
299 if (!OwnerActor->HasAuthority())
300 {
301 return nullptr;
302 }
303
304 // Get item definition instance
305 UItemDefinition* CachedDefinition = OwnerComponent->GetCachedDefinition(DefinitionClass);
306 const UItemFragment_Storable* StorableFragment = CachedDefinition->FindFragmentByClass<UItemFragment_Storable>();
307 if (!IsValid(StorableFragment))
308 {
309 return nullptr;
310 }
311
312 // Create a new entry
313 FInventoryEntry& Entry = Entries.AddDefaulted_GetRef();
314 const int32 Index = Entries.Num();
315
316 Entry.Instance = NewObject<UItemInstance>(OwnerActor);
317 Entry.Instance->SetDefinition(CachedDefinition);
318
319 Entry.StackCount = StorableFragment->CanStack() ? FMath::Min(Count, StorableFragment->MaxStackCount) : 1;
320 Count -= Entry.StackCount;
321
322 for (auto Fragment : CachedDefinition->Fragments)
323 {
324 if (IsValid(Fragment))
325 {
326 Fragment->OnInstanceCreated(Entry.Instance);
327 }
328 }
329
330 Internal_OnEntryAdded(Index, Entry);
331 MarkItemDirty(Entry);
332
333 return Entry.Instance;
334}

336bool FInventoryList::CanAdd(const TSubclassOf<UItemDefinition>& DefinitionClass, FGameplayTag& OutFailureReason, const int32 InCount)
337{
338 UItemDefinition* CachedDefinition = OwnerComponent->GetCachedDefinition(DefinitionClass);
339 if (!IsValid(CachedDefinition))
340 {
341 UE_LOG(LogInventorySystem, Warning, TEXT("Tried to instantiate an item instance from invalid definition class."));
342 OutFailureReason = InventorySystemGameplayTags::TAG_Inventory_Failure_NullDefinition;
343 return false;
344 }
345
346 if (!CachedDefinition->CanBeGiven(OwnerComponent, OutFailureReason))
347 {
348 UE_LOG(LogInventorySystem, Warning, TEXT("Tried to instantiate an item instance from %s, but the definition conditions are not met."), *GetNameSafe(DefinitionClass));
349 if (!OutFailureReason.IsValid())
350 OutFailureReason = InventorySystemGameplayTags::TAG_Inventory_Failure_DefinitionRefused;
351 return false;
352 }
353
354 // If the object has no storable fragment, it cannot be added to the inventory
355 const UItemFragment_Storable* StorableFragment = CachedDefinition->FindFragmentByClass<UItemFragment_Storable>();
356 if (!IsValid(StorableFragment))
357 {
358 UE_LOG(LogInventorySystem, Warning, TEXT("Tried to instantiate an item instance of %s but do not own UItemFragment_Storable fragment."), *GetNameSafe(CachedDefinition));
359 OutFailureReason = InventorySystemGameplayTags::TAG_Inventory_Failure_NotStorable;
360 return false;
361 }
362
363 // Check if the object is unique
364 if (StorableFragment->IsUnique())
365 {
366 if (const FInventoryEntryHandle Handle = FindHandleOfType(DefinitionClass); Handle.IsValid())
367 {
368 // Allow if we can fit in the existing stack
369 if (StorableFragment->CanStack() && (StorableFragment->MaxStackCount - Handle.StackCount <= InCount))
370 return true;
371
372 UE_LOG(LogInventorySystem, Warning, TEXT("Tried to instantiate a single item instance %s but another instance already exists."), *GetNameSafe(CachedDefinition));
373 OutFailureReason = InventorySystemGameplayTags::TAG_Inventory_Failure_Uniqueness;
374 return false;
375 }
376 }
377 return true;
378}
