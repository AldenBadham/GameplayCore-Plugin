16void UAbilitySet::GiveToAbilitySystem(UAbilitySystemComponent* AbilitySystemComp, FAbilitySetHandles* Handles, UObject* SourceObject) const
17{
18 check(AbilitySystemComp);
19 if (!AbilitySystemComp->IsOwnerActorAuthoritative()) UE_LOG(LogAbilitySystem, Error, TEXT("Must be authoritive to give or take ability sets"));
20
21 // Grant the attribut sets
22 for (int Index = 0; Index < AttributeSets.Num(); ++Index)
23 {
24 const TSubclassOf<UAttributeSet>& AttributeSetClass = AttributeSets[Index];
25
26 UE_CLOG(!IsValid(AttributeSetClass), LogAbilitySystem, Error, TEXT("Attributes[%d] on ability set [%s] is not valid"), Index, *GetNameSafe(this));
27
28 UAttributeSet* AttributeSet = NewObject<UAttributeSet>(AbilitySystemComp, AttributeSetClass);
29 AbilitySystemComp->AddAttributeSetSubobject(AttributeSet);
30
31 if (Handles)
32 {
33 Handles->AddAttributeSet(AttributeSet);
34 }
35 }
36
37 // Grant the gameplay abilities
38 for (int Index = 0; Index < GameplayAbilities.Num(); ++Index)
39 {
40 const FAbilitySet_GameplayAbility& GameplayAbilitySet = GameplayAbilities[Index];
41
42 UE_CLOG(!IsValid(GameplayAbilitySet.Ability), LogAbilitySystem, Error, TEXT("Attributes[%d] on ability set [%s] is not valid"), Index, *GetNameSafe(this));
43
44 UGameplayAbilityBase* AbilityCDO = GameplayAbilitySet.Ability->GetDefaultObject<UGameplayAbilityBase>();
45
46 FGameplayAbilitySpec AbilitySpec(AbilityCDO, GameplayAbilitySet.AbilityLevel);
47 AbilitySpec.SourceObject = SourceObject;
48 AbilitySpec.GetDynamicSpecSourceTags().AddTag(GameplayAbilitySet.InputTag);
49
50 const FGameplayAbilitySpecHandle AbilitySpecHandle = AbilitySystemComp->GiveAbility(AbilitySpec);
51
52 if (Handles)
53 {
54 Handles->AddAbilitySpecHandle(AbilitySpecHandle);
55 }
56 }
57
58 // Grant the gameplay effects.

60 {
61 const FAbilitySet_GameplayEffect& GameplayEffectSet = GameplayEffects[Index];
62
63 UE_CLOG(!IsValid(GameplayEffectSet.GameplayEffect), LogAbilitySystem, Error, TEXT("Attributes[%d] on ability set [%s] is not valid"), Index, *GetNameSafe(this));
64
65 const UGameplayEffect* GameplayEffect = GameplayEffectSet.GameplayEffect->GetDefaultObject<UGameplayEffect>();
66 const FActiveGameplayEffectHandle GameplayEffectHandle = AbilitySystemComp->ApplyGameplayEffectToSelf(GameplayEffect, GameplayEffectSet.EffectLevel, AbilitySystemComp->MakeEffectContext());
67
68 if (Handles)
69 {
70 Handles->AddGameplayEffectHandle(GameplayEffectHandle);
71 }
72 }
