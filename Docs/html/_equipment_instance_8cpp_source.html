118void UEquipmentInstance::SpawnActors(const TArray<FEquipmentActorSet>& ActorsToSpawn)
119{
120 UWorld* World = GetWorld();
121 verifyf(IsValid(World), TEXT("Invalid world to spawn actors for %s"), *GetName());
122
123 if (APawn* OwningPawn = GetPawn(); IsValid(OwningPawn))
124 {
125 // Use a root scene component as an attachment target by default
126 USceneComponent* AttachTarget = OwningPawn->GetRootComponent();
127 if (const ACharacter* Character = Cast<ACharacter>(OwningPawn))
128 {
129 AttachTarget = Character->GetMesh();
130 }
131
132 for (const auto& [SlotTag, ActorClass, AttachSocket, AttachTransform] : ActorsToSpawn)
133 {
134 if (!IsValid(ActorClass))
135 {
136 UE_LOG(LogEquipmentSystem, Warning, TEXT("Tried to spawn equipment actors with an invalid actor class for %s!"), *GetName());
137 continue;
138 }
139
140 if (const auto NewActor = World->SpawnActorDeferred<AActor>(ActorClass, FTransform::Identity, OwningPawn); IsValid(NewActor))
141 {
142 NewActor->SetInstigator(OwningPawn);
143 NewActor->FinishSpawning(FTransform::Identity, true);
144
145 // Set relative transform before attaching
146 NewActor->SetActorRelativeTransform(AttachTransform);
147 NewActor->AttachToComponent(AttachTarget, FAttachmentTransformRules::KeepRelativeTransform, AttachSocket);
148
149 SpawnedActors.Add(NewActor);
150 }
151 }
152 return;
153 }
154
155 UE_LOG(LogEquipmentSystem, Error, TEXT("Tried to spawn equipment actors with an invalid pawn !"));
156}
