26void UItemDefinition::PostEditChangeProperty(FPropertyChangedEvent& PropertyChangedEvent)
27{
28 Super::PostEditChangeProperty(PropertyChangedEvent);
29
30 if (PropertyChangedEvent.Property && PropertyChangedEvent.GetPropertyName() == GET_MEMBER_NAME_CHECKED(UItemDefinition, Fragments))
31 {
32 // Check Fragment rules from the Inventory System Settings
33 const UInventorySystemSettings* InventorySettings = GetDefault<UInventorySystemSettings>();
34 UE_CLOG(IsValid(InventorySettings), LogInventorySystem, Error, TEXT("Invalid Inventory System Settings class get."));
35
36 // Identify newly added fragments
37 TMap<int32, UItemFragment*> NewFragments;
38 TArray<TSubclassOf<UItemFragment>> ExistingClasses;
39
40 for (int32 Index = 0; Index < Fragments.Num(); ++Index)
41 {
42 if (UItemFragment* const Fragment = Fragments[Index]; IsValid(Fragment))
43 {
44 if (!PreviousFragments.Contains(Fragment))
45 {
46 NewFragments.Add(Index, Fragment);
47 continue;
48 }
49 ExistingClasses.Add(Fragment->GetClass());
50 }
51 }
52
53 // Quick duplicate check
54 for (auto& [Index, NewFragment] : NewFragments)
55 {
56 TSubclassOf<UItemFragment> NewFragmentClass = NewFragment->GetClass();
57 if (ExistingClasses.Contains(NewFragmentClass))
58 {
59 FString Context = FString::Printf(TEXT("Duplicated item fragment on %s."), *this->GetName());
60
61 FNotificationInfo Info(FText::FromString(Context));
62 FSlateNotificationManager::Get().AddNotification(Info)->SetCompletionState(SNotificationItem::CS_Fail);
63
64 UE_LOG(LogInventorySystem, Error, TEXT("Duplicated item fragment %s of %s."), *NewFragmentClass->GetName(), *this->GetName());
65
66 if (Fragments.IsValidIndex(Index))
67 {
68 Fragments[Index] = nullptr;
69 }
70 }
71 }
72
73 // Check rule validity for new fragments
74 for (auto& [Index, NewFragment] : NewFragments)
75 {
76 TSubclassOf<UItemFragment> NewFragmentClass = NewFragment->GetClass();
77 if (const FItemFragmentRule* Rule = InventorySettings->FindRuleForClass(NewFragmentClass))
78 {
79 if (FString ErrorMessage; !Rule->IsRuleSatisfied(ExistingClasses, ErrorMessage))
80 {
81 FString Context = FString::Printf(TEXT("Fragment rule not filled on %s."), *this->GetName());
82
83 FNotificationInfo Info(FText::FromString(Context));
84 FSlateNotificationManager::Get().AddNotification(Info)->SetCompletionState(SNotificationItem::CS_Fail);
85
86 UE_LOG(LogInventorySystem, Error, TEXT("Fragment rule not filled for fragment %s of %s. %s"), *NewFragmentClass->GetName(), *this->GetName(), *ErrorMessage);
87
88 if (Fragments.IsValidIndex(Index))
89 {
90 Fragments[Index] = nullptr;
91 }
92 }
93 }
94 }
95
96 // Previous fragments update
97 PreviousFragments = Fragments;
98
99 PostEditChange();
100 }
101}
