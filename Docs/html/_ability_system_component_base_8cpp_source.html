78void UAbilitySystemComponentBase::ProcessAbilityInput(float DeltaTime, bool bGamePaused)
79{
80 // Check if abilities input activation is blocked
81 if (HasMatchingGameplayTag(AbilitySystemCoreTags::TAG_Gameplay_AbilitiesBlocked))
82 {
83 ClearAbilityInput();
84 return;
85 }
86
87 static TArray<FGameplayAbilitySpecHandle> AbilitiesToActivate;
88 AbilitiesToActivate.Reset();
89
90 // Process all abilities that activate when the input is held.
91 for (const FGameplayAbilitySpecHandle& SpecHandle : InputHeldSpecHandles)
92 {
93 if (const FGameplayAbilitySpec* AbilitySpec = FindAbilitySpecFromHandle(SpecHandle))
94 {
95 if (AbilitySpec->Ability && !AbilitySpec->IsActive())
96 {
97 const UGameplayAbilityBase* BaseAbilityCDO = Cast<UGameplayAbilityBase>(AbilitySpec->Ability);
98 if (BaseAbilityCDO && BaseAbilityCDO->GetActivationPolicy() == EAbilityActivationPolicy::WhileInputActive)
99 {
100 AbilitiesToActivate.AddUnique(AbilitySpec->Handle);
101 }
102 }
103 }
104 }
105
106 // Process all abilities that had their input pressed this frame.
107 for (const FGameplayAbilitySpecHandle& SpecHandle : InputPressedSpecHandles)
108 {
109 if (FGameplayAbilitySpec* AbilitySpec = FindAbilitySpecFromHandle(SpecHandle))
110 {
111 if (AbilitySpec->Ability)
112 {
113 AbilitySpec->InputPressed = true;
114
115 if (AbilitySpec->IsActive())
116 {
117 // Ability is active so pass along the input event.
118 AbilitySpecInputPressed(*AbilitySpec);
119 }
120 else
121 {
122 const UGameplayAbilityBase* BaseAbilityCDO = Cast<UGameplayAbilityBase>(AbilitySpec->Ability);
123
124 if (BaseAbilityCDO && BaseAbilityCDO->GetActivationPolicy() == EAbilityActivationPolicy::OnInputTriggered)
125 {
126 AbilitiesToActivate.AddUnique(AbilitySpec->Handle);
127 }
128 }
129 }
130 }
131 }
132
133 // Try to activate all the abilities that are from presses and holds.
134 // We do it all at once so that held inputs don't activate the ability
135 // and then also send a input event to the ability because of the press.
136 for (const FGameplayAbilitySpecHandle& AbilitySpecHandle : AbilitiesToActivate)
137 {
138 TryActivateAbility(AbilitySpecHandle);
139 }
140
141 // Process all abilities that had their input released this frame.
142 for (const FGameplayAbilitySpecHandle& SpecHandle : InputReleasedSpecHandles)
143 {
144 if (FGameplayAbilitySpec* AbilitySpec = FindAbilitySpecFromHandle(SpecHandle))
145 {
146 if (AbilitySpec->Ability)
147 {
148 AbilitySpec->InputPressed = false;
149
150 if (AbilitySpec->IsActive())
151 {
152 // Ability is active so pass along the input event.
153 AbilitySpecInputReleased(*AbilitySpec);
154 }
155 }
156 }
157 }
158
159 // Clear the cached ability handles.
160 InputPressedSpecHandles.Reset();
161 InputReleasedSpecHandles.Reset();
162}

179void UAbilitySystemComponentBase::TryActivateAbilitiesOnSpawn()
180{
181 // Stop from removing ability from the ability system to iterate through the abilities
182 ABILITYLIST_SCOPE_LOCK();
183
184 for (const FGameplayAbilitySpec& AbilitySpec : ActivatableAbilities.Items)
185 {
186 const UGameplayAbilityBase* Ability = CastChecked<UGameplayAbilityBase>(AbilitySpec.Ability);
187 Ability->TryActivateAbilityOnSpawn(AbilityActorInfo.Get(), AbilitySpec);
188 }

191void UAbilitySystemComponentBase::AbilitySpecInputPressed(FGameplayAbilitySpec& Spec)
192{
193 Super::AbilitySpecInputPressed(Spec);
194
195 // UE : We don't support UGameplayAbility::bReplicateInputDirectly.
196 // Use replicated events instead so that the WaitInputPress ability task works.
197 if (Spec.IsActive())
198 {
199 FPredictionKey OriginalPredictionKey;
200 if (const UGameplayAbility* Instance = Spec.GetPrimaryInstance())
201 {
202 OriginalPredictionKey = Instance->GetCurrentActivationInfo().GetActivationPredictionKey();
203 }
204
205 // Invoke the InputPressed event. This is not replicated here. If someone is listening, they may replicate the InputPressed event to the server.
206 InvokeReplicatedEvent(EAbilityGenericReplicatedEvent::InputPressed, Spec.Handle, OriginalPredictionKey);
207 }
208}
