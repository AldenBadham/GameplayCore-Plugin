57void FEquipmentList::PostReplicatedChange(const TArrayView<int32> ChangedIndices, int32 FinalSize)
58{
59 for (const int32 Index : ChangedIndices)
60 {
61 if (Entries.IsValidIndex(Index))
62 {
63 FEquipmentEntry& Entry = Entries[Index];
64
65 if (IsValid(Entry.Instance))
66 {
67 Entry.Instance->OnEquipped();
68 }
69
70 if (Entry.Instance != Entry.LastInstance.Get() && Entry.LastInstance.IsValid())
71 {
72 Entry.LastInstance->OnUnequipped();
73 Entry.LastInstance = MakeWeakObjectPtr<UEquipmentInstance>(Entry.Instance);
74 }
75
76 Internal_OnEntryChanged(Index, Entry);
77 }
78 }

81FEquipmentResult FEquipmentList::Add(const TSubclassOf<UEquipmentDefinition>& DefinitionClass, UItemInstance* SourceItemInstance)
82{
83 FEquipmentResult Result;
84 if (DefinitionClass == nullptr || !IsValid(OwnerComponent))
85 {
86 UE_LOG(LogEquipmentSystem, Warning, TEXT("Tried to instantiate an equipment instance from invalid definition class."));
87 Result.FailureReason = EquipmentSystemGameplayTags::TAG_Equipment_Failure_MissingDefinition;
88 return Result;
89 }
90
91 AActor* OwnerActor = OwnerComponent->GetOwner();
92 if (!OwnerActor->HasAuthority())
93 {
94 UE_LOG(LogEquipmentSystem, Warning, TEXT("Tried to instantiate an equipment instance from invalid definition class."));
95 Result.FailureReason = EquipmentSystemGameplayTags::TAG_Equipment_Failure_NotAuthority;
96 return Result;
97 }
98
99 UEquipmentDefinition* CachedDefinition = OwnerComponent->GetCachedDefinition(DefinitionClass);
100 if (!CachedDefinition->CanBeEquipped(OwnerComponent, Result.FailureReason))
101 {
102 UE_LOG(LogEquipmentSystem, Warning, TEXT("Tried to instantiate an equipment instance from %s, but the definition conditions are not met."), *GetNameSafe(DefinitionClass));
103 if (!Result.FailureReason.IsValid())
104 Result.FailureReason = EquipmentSystemGameplayTags::TAG_Equipment_Failure_DefinitionRefused;
105 return Result;
106 }
107
108 // Prepare an instance type to spawn, use the default one by default
109 TSubclassOf<UEquipmentInstance> InstanceType = CachedDefinition->InstanceClass;
110 if (InstanceType == nullptr)
111 {
112 InstanceType = UEquipmentDefinition::StaticClass();
113 }
114
115 // Add default entry
116 FEquipmentEntry& Entry = Entries.AddDefaulted_GetRef();
117
118 Entry.EquipmentDefinition = DefinitionClass;
119 Entry.Instance = NewObject<UEquipmentInstance>(OwnerActor, InstanceType);
120
121 UEquipmentInstance* Instance = Entry.Instance;
122 Instance->SetDefinition(CachedDefinition);
123 Instance->SetInstigator(OwnerActor);
124 Instance->SetSourceItem(SourceItemInstance);
125
126 Result.Instance = Instance;
127
128 // Give the ability sets
129 if (UAbilitySystemComponent* AbilitySystemComp = GetAbilitySystemComponent())
130 {
131 for (const TObjectPtr<const UAbilitySet>& AbilitySet : CachedDefinition->AbilitySets)
132 {
133 AbilitySet->GiveToAbilitySystem(AbilitySystemComp, &Entry.Handles, Instance);
134 }
135 }
136
137 // Ask the instance to spawn attachment actors
138 Instance->SpawnActors(CachedDefinition->ActorsToSpawn);
139 Instance->OnSpawned();
140
141 // Mark the item dirty for the serializer replication
142 MarkItemDirty(Entry);
143
144 return Result;
145}

147void FEquipmentList::Remove(UEquipmentInstance* Instance, FGameplayTag& OutFailureReason)
148{
149 for (auto EntryIterator = Entries.CreateIterator(); EntryIterator; ++EntryIterator)
150 {
151 if (FEquipmentEntry& Entry = *EntryIterator; Entry.Instance == Instance)
152 {
153 if (UAbilitySystemComponent* AbilitySystemComp = GetAbilitySystemComponent())
154 {
155 Entry.Handles.TakeFromAbilitySystem(AbilitySystemComp);
156 }
157
158 // Ask the instance to destroy its attachment actors
159 Instance->DestroyActors();
160
161 EntryIterator.RemoveCurrent();
162 MarkArrayDirty();
163 }
164 }
